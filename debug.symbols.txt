# Debug Symbols for Momentum Template
# This file contains all variable replacements and symbols used in the template

## Component Parameters
aspire=__ASPIRE__
api=__API__
back-office=__BACK_OFFICE__
docs=__DOCS__
orleans=__ORLEANS__

## Configuration Parameters
db-config=__DB_CONFIG__
kafka=__KAFKA__
port=__SERVICE_BASE_PORT__
org=__ORG__

## Content Parameters
no-sample=__NO_SAMPLE__
project-only=__PROJECT_ONLY__

## Library Parameters
libs=__LIBS__
lib-name=__LIB_NAME__
mmt-version=__TEMPLATE_MOMENTUM_VERSION__
debug-symbols=__DEBUG_SYMBOLS__

## Generated Symbols
DefaultOrgName -> Generated join of name + " Team"
OrgName=OrgName -> Coalesced organization name from org parameter or DefaultOrgName
OrgNameKebab=org-name -> Organization name in kebab-case
AppDomainSnakeCase=app_domain -> Project name in snake_case (fileRename: app_domain)
AppDomainKebab=app-domain -> Project name in kebab-case (fileRename: app-domain)
AppDomainCamelCase=appDomain -> Project name in camelCase (fileRename: appDomain)
